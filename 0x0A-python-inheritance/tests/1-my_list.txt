from 1-my_list import MyList

# Create an instance of MyList
my_list = MyList([3, 1, 2])

# Test the print_sorted method
my_list.print_sorted()
# Expected output:
# [1, 2, 3]

# Test with an empty list
empty_list = MyList()
empty_list.print_sorted()
# Expected output:
# []

# Test with a list with one element
single_element_list = MyList([1])
single_element_list.print_sorted()
# Expected output:
# [1]

# Test with a list with multiple elements, already sorted
sorted_list = MyList([1, 2, 3])
sorted_list.print_sorted()
# Expected output:
# [1, 2, 3]

# Test with a list with multiple elements, reverse sorted
reverse_sorted_list = MyList([3, 2, 1])
reverse_sorted_list.print_sorted()
# Expected output:
# [1, 2, 3]

# Test with a list with repeated elements
repeated_elements_list = MyList([3, 1, 2, 1, 3])
repeated_elements_list.print_sorted()
# Expected output:
# [1, 1, 2, 3, 3]

# Test with a list with negative elements
negative_elements_list = MyList([-1, -3, -2])
negative_elements_list.print_sorted()
# Expected output:
# [-3, -2, -1]
